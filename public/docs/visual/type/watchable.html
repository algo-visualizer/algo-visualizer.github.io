 23class Watchable(ABC):
 24
 25    def __init__(self, var: str, *, expr: bool = False):
 26        """
 27        **Provide variable names or expressions but not both.** `expr=False` looks up `vars` arguments by name at locals/globals of the execution frame which has better performance. `expr=True` treats `vars` arguments as expressions and directly evaluate them by using `eval()` during capture.
 28
 29        **Use `expr=False` in most cases.** Use `expr=True` when you must watch a composite expression or computation that is not stored in a simple variable.
 30
 31        Examples:
 32            ```python
 33            Watchable('i')
 34            Watchable('i + j', expr=True) # The varname will be 'i + j' in this case
 35            Watchable('i, j', expr=True) # Notice that it's not Watchable('i, j')
 36
 37            # Wrong usage
 38            Watchable()
 39            ```
 40        """
 41
 42        if not isinstance(var, str):
 43            raise TypeError("`var` must be a string")
 44        
 45        if not isinstance(expr, bool):
 46            raise TypeError("`expr` must be a boolean")
 47
 48        if not expr:
 49            self.varname = var
 50            self._expr = None
 51        else:
 52            self.varname = var
 53            self._expr = var
 54
 55    def __eq__(self, other):
 56        return type(self) == type(other) and self.varname == other.varname
 57    
 58    def __hash__(self):
 59        return hash((type(self), self.varname))
 60
 61    def capture(self, frame: FrameType | None = None):
 62        """
 63        Capture the current state of the watchable variable from the given frame
 64        Args:
 65            frame (FrameType | None): The execution frame. If None, it uses the parent frame (the frame which called this function).
 66        Returns:
 67            Any: The captured value of the watchable variable, or _NotCaptured if not found.
 68        """
 69
 70        if frame is None:
 71            frame = _get_caller_frame(currentframe())
 72
 73        local_vars = frame.f_locals
 74        global_vars = frame.f_globals
 75
 76        if self._expr:
 77            try:
 78                value = eval(self._expr, global_vars, local_vars)
 79            except Exception:
 80                return _NotCaptured()
 81            return value
 82        
 83        # Capture Watched Variable (Look in Locals then Globals)
 84        # This matches names strictly, no expression evaluation.
 85        found = False
 86        value = None
 87        
 88        if self.varname in local_vars:
 89            value = local_vars[self.varname]
 90            found = True
 91        elif self.varname in global_vars:
 92            value = global_vars[self.varname]
 93            found = True
 94
 95        if not found:
 96            return _NotCaptured()
 97
 98        return value

273        var: str,
274        head_id: Callable[[_NodesType], _IdType], 
275        next_ids: Callable[[_NodesType, _IdType], _IdType | tuple[_IdType, _WeightType] | list[_IdType] | list[tuple[_IdType, _WeightType]]],
276        value: Callable[[_NodesType, _IdType], _NodeType],

278        expr: bool = False,

282        self._next_ids_func = next_ids

285    def _traverse(self, nodes: _NodesType):
286        head_id = self._head_id_func(nodes)
287        visited: set[_IdType] = set([head_id])
288        queue: deque[_IdType] = deque([head_id])
289        graph_content = NodesGraphContent()
290        rawids: list[_IdType] = []
291
292        while queue:
293            id = queue.popleft()
294            rawids.append(id)
295            nexts = self._next_ids_func(nodes, id)
296            if not isinstance(nexts, list):
297                nexts = [nexts]
298            new_nexts: list[tuple[_IdType, _WeightType]] = []
299            nexts_pure_str: list[tuple[str, str]] = []
300            for n in nexts:
301                if not isinstance(n, tuple):
302                    n = (n, cast(_WeightType, 1))
303                if n[0] is None: continue
304                new_nexts.append(n)
305                nexts_pure_str.append((repr(n[0]), repr(n[1])))
306            nexts = new_nexts
307            graph_content.value[repr(id)] = NodesGraphContentItem(
308                value="", # Get value in self.generate_graph 
309                nexts=nexts_pure_str
310            )
311            # traverse its neighbors
312            for neighborid, _ in nexts:
313                if neighborid in visited:
314                    continue
315                visited.add(neighborid)
316                queue.append(neighborid)
317
318        return graph_content, rawids
319
320    @override
321    def generate_graph(self, frame = None) -> NodesGraph:
322        if frame is None:
323            frame = _get_caller_frame(currentframe())
324        curr_value = self.capture(frame)
325        frameid, parent_frameid = str(id(frame)), None
326        if parent_frame := frame.f_back is not None:
327            parent_frameid = str(id(parent_frame))
328        if isinstance(curr_value, _NotCaptured):
329            return NodesGraph(
330                frameid=frameid,
331                parent_frameid=parent_frameid,
332                notcaptured=True,
333                content=repr(curr_value),
334            )
335        curr_value = cast(_NodesType, curr_value)
336        graph_content, rawids = self._traverse(curr_value)
337        for var in self._index_registry.values():
338            index = var.capture(frame)
339            pointer = Pointer(
340                name=var.varname, index=repr(index), 
341                notcaptured=True if isinstance(index, _NotCaptured) else False
342            )
343            graph_content.pointers.append(pointer)
344        itemvars_value = {var.varname: var.capture(frame) for var in self._item_registry.values()}
345        for nodeid in rawids:
346            nodevalue = self._value_func(curr_value, nodeid)
347            for itemvarname, itemvar in itemvars_value.items():
348                if nodevalue is not itemvar:
349                    continue
350                graph_content.pointers.append(Pointer(
351                    name=itemvarname, index=repr(nodeid), 
352                    notcaptured=False
353                ))
354            graph_content.value[repr(nodeid)].value = repr(nodevalue)
355        return NodesGraph(
356            frameid=frameid,
357            parent_frameid=parent_frameid,
358            notcaptured=False,
359            content=graph_content,
360        )
